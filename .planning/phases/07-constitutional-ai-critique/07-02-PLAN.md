       class CritiqueEngine:
         __init__(self, constitution: ConstitutionConfig, guardrail_engine: GuardrailEngine):
           Store constitution and guardrail_engine. Build rails_config dict from guardrail_engine._rails_config for critique_threshold lookup.

         async def run_critique_loop(self, response_data: dict, output_results: list[RailResult], request_model: str, http_client, pii_strictness: str = "balanced") -> dict | None:
           - Find the first output rail where score >= rail's critique_threshold (from self._guardrail_engine._rails_config[rail].critique_threshold) AND the overall output was NOT blocked (score < threshold).
           - If no rail exceeds critique_threshold: return None.
           - Resolve judge_model: if constitution.judge_model == "default" use request_model, else use constitution.judge_model.
           - Get relevant principles: filter constitution.principles by enabled=True AND category in RAIL_TO_CATEGORIES[triggering_rail].
           - Build critique prompt via _build_critique_prompt().
           - Call _call_judge() with 60s asyncio timeout. On timeout or exception: return None (fail-open).
           - Parse judge JSON response. Expected shape: {"violated_principles": [...], "critique": "...", "revision": "...", "confidence": 0.0-1.0}.
           - If parse fails: return None (fail-open).
           - PII-redact the revision text: from harness.pii.redactor import redact; revision = redact(judge_response["revision"], pii_strictness).
           - Re-check the revision against output rails: call guardrail_engine.check_output() with a synthetic response_data containing the revision text. Use the revision_decision.all_results to get the revision_score for the triggering rail.
           - If revision still scores >= critique_threshold for the triggering rail: outcome = "fallback_hard_block", fallback_hard_block = True.
           - Else: outcome = "revised", fallback_hard_block = False.
           - Return structured cai_critique dict:
             ```python
             {
                 "triggered_by": triggering_rail_name,
                 "judge_model": resolved_judge_model_id,
                 "original_score": triggering_score,
                 "critique_threshold": critique_threshold_value,
                 "judge_response": {
                     "violated_principles": judge_response["violated_principles"],
                     "critique": judge_response["critique"],
                     "revision": redacted_revision_text,
                     "confidence": judge_response["confidence"],
                 },
                 "revision_score": revision_score_for_triggering_rail,
                 "outcome": outcome,
                 "fallback_hard_block": fallback_hard_block,
             }
             ```

         def _build_critique_prompt(self, output_text: str, triggering_rail: str, relevant_principles: list[Principle]) -> tuple[str, str]:
           Returns (system_prompt, user_content).
           system_prompt: "You are a safety judge. Evaluate the following AI-generated text against the listed constitutional principles. Respond ONLY with valid JSON matching the schema: {\"violated_principles\": [<principle IDs>], \"critique\": \"<explanation>\", \"revision\": \"<revised text>\", \"confidence\": <0.0-1.0>}. The revision must address all identified violations while preserving the original intent as much as possible."
           user_content: "Constitutional principles:\n{principle_list}\n\nTriggering guardrail: {triggering_rail}\n\nAI-generated text to evaluate:\n{output_text}"
           principle_list: sorted by priority descending, formatted as "- [{p.id}] {p.text}" for each enabled principle.

         async def _call_judge(self, http_client, model: str, system_prompt: str, user_content: str) -> dict:
