    compute_metrics(cases: list[dict], results: list[dict]) -> dict:
    - cases have fields: prompt, expected_action ("block"|"allow"|"steer"), category, description

    - Return dict with keys: f1, precision, recall, correct_refusal_rate, false_refusal_rate, total_cases, per_category
    - per_category is dict mapping category name to {tp, fp, tn, fn} counts
    - Round all float metrics to 4 decimal places

    compute_latency_percentiles(latencies: list[int]) -> dict:
    - Return {"p50": value, "p95": value}
    - Use sorted list + index calculation: p50 = sorted[len//2], p95 = sorted[int(len*0.95)]

    async def run_replay(
        dataset_path: str,
        gateway_base_url: str,
        api_key: str,
        trace_store: TraceStore,
        model: str = "llama3.1",
    ) -> dict:
    - Load JSONL: read file lines, json.loads each non-empty line -> cases list
    - Generate run_id = f"replay-{uuid.uuid4().hex[:12]}"
    - Record run_start = datetime.now(timezone.utc).isoformat()
    - Iterate cases with httpx.AsyncClient(base_url=gateway_base_url, timeout=60.0):
      - For each case: POST /v1/chat/completions with headers={"Authorization": f"Bearer {api_key}"}, json={"model": model, "messages": [{"role": "user", "content": case["prompt"]}]}
      - Measure latency_ms = int((time.monotonic() - t0) * 1000)
      - Determine actual_action: "block" if resp.status_code in (400, 403, 422) else "allow"
      - Append to results list: {"case": case, "actual_action": actual_action, "latency_ms": latency_ms, "status_code": resp.status_code}
    - After all cases: batch-read traces by timerange (run_start to now) for guardrail_decisions detail
    - Compute metrics via compute_metrics(cases, results)
    - Compute latency via compute_latency_percentiles([r["latency_ms"] for r in results])
    - Compute critique_trigger_rate: count traces with non-null cai_critique / total_cases
    - Build config_snapshot: {"model": model, "dataset": dataset_path, "gateway": gateway_base_url, "timestamp": run_start}
    - Store via await trace_store.write_eval_run({"run_id": run_id, "timestamp": run_start, "source": "replay", "metrics": {**metrics_dict, **latency_dict, "critique_trigger_rate": rate}, "config_snapshot": config_snapshot})
